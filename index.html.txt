// ── TODAY ─────────────────────────────────────────────────
async function loadToday(){
  const el = document.getElementById('today-content');
  el.innerHTML = '<div class="loader"><div class="spinner"></div>Yuklanmoqda...</div>';
  try{
    const [tod, bal] = await Promise.all([
      fetch(API+'/today').then(r=>r.json()),
      fetch(API+'/balance').then(r=>r.json())
    ]);
    const chU = tod.total_ch_usd, chZ = tod.total_ch_uzs;
    const kiU = tod.total_ki_usd, kiZ = tod.total_ki_uzs;
    let html = `
      <div class="card">
        <div class="label">💰 Joriy balans</div>
        <div class="value blue">${bal.formatted}</div>
      </div>
      <div class="grid2">
        <div class="stat-card">
          <div class="stat-label">📤 Bugungi chiqim</div>
          <div class="stat-value red">${sumStr(chU,chZ)}</div>
        </div>
        <div class="stat-card">
          <div class="stat-label">📥 Bugungi kirim</div>
          <div class="stat-value green">${sumStr(kiU,kiZ)}</div>
        </div>
      </div>
      <div class="date-badge">📅 ${tod.date}</div>
    `;
    if(tod.chiqimlar.length){
      html += '<div class="section-title">Chiqimlar</div><div class="tx-list">';
      tod.chiqimlar.forEach(t=>{
        html += txItem(t,'CHIQIM');
      });
      html += '</div>';
    }
    if(tod.kirimlar.length){
      html += '<div class="section-title">Kirimlar</div><div class="tx-list">';
      tod.kirimlar.forEach(t=>{
        html += txItem(t,'KIRIM');
      });
      html += '</div>';
    }
    if(!tod.chiqimlar.length && !tod.kirimlar.length){
      html += '<div class="empty"><div class="empty-icon">🌙</div>Bugun hech narsa yo\'q</div>';
    }
    el.innerHTML = html;
  }catch(e){
    el.innerHTML = '<div class="empty">❌ Xato: '+e.message+'</div>';
  }

// ── FORM ──────────────────────────────────────────────────
function initForm(){
  const today = new Date().toLocaleDateString('en-CA');
  document.getElementById('f-sana').value = today;
  document.getElementById('f-summa').value = '';
  document.getElementById('f-note').value = '';
  renderTurs();
}

function selType(type, btn){
  state.type = type; state.tur = '';
  document.querySelectorAll('#p-add .btn-grid.col2 .sel-btn').forEach(b=>b.classList.remove('active'));
  btn.classList.add('active');
  document.getElementById('tur-label').textContent = type==='CHIQIM'?'Chiqim turi':'Kirim turi';
  renderTurs();
}

function renderTurs(){
  const turs = state.type==='CHIQIM'?CHIQIM_TURS:KIRIM_TURS;
  const grid = document.getElementById('tur-grid');
  grid.className = 'btn-grid '+(turs.length>6?'col3':'col2');
  grid.innerHTML = turs.map(t=>`
    <button class="sel-btn${state.tur===t?' active':''}" onclick="selTur('${t}',this)">
      ${EMOJIS[t]||''} ${t}
    </button>`).join('');
}

function selTur(t, btn){
  state.tur = t;
  document.querySelectorAll('#tur-grid .sel-btn').forEach(b=>b.classList.remove('active'));
  btn.classList.add('active');
}

function selBtn(key, btn){
  const parent = btn.parentElement;
  parent.querySelectorAll('.sel-btn').forEach(b=>b.classList.remove('active'));
  btn.classList.add('active');
  const txt = btn.textContent.trim();
  if(key==='egasi') state.egasi = txt.includes('Ferudin')?'FERUDIN':'GULOYIM';
  if(key==='tolov') state.tolov = txt.includes('Cash')?'CASH':txt.includes('Card')?'CARD':'OTHER';
  if(key==='valyuta') state.valyuta = txt.includes('USD')?'USD':'UZS';
}

async function submitForm(){
  if(!state.tur){ alert('Tur tanlang!'); return; }
  const summa = parseFloat(document.getElementById('f-summa').value);
  if(!summa||summa<=0){ alert('Summa kiriting!'); return; }
  const sanaRaw = document.getElementById('f-sana').value;
  if(!sanaRaw){ alert('Sanani kiriting!'); return; }
  const [y,m,d] = sanaRaw.split('-');
  const sana = `${d}.${m}.${y}`;
  const body = {
    type:state.type, sana, egasi:state.egasi,
    tur:state.tur, tolov:state.tolov,
    valyuta:state.valyuta, summa,
    note:document.getElementById('f-note').value||''
  };
  const btn = document.querySelector('#p-add .submit-btn');
  btn.textContent='⏳ Saqlanmoqda...'; btn.disabled=true;
  try{
    const r = await fetch(API+'/transaction',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify(body)});
    const res = await r.json();
    if(res.success){
      btn.textContent='✅ Saqlandi!';
      setTimeout(()=>{btn.textContent='✅ Saqlash';btn.disabled=false;},2000);
      document.getElementById('f-summa').value='';
      document.getElementById('f-note').value='';
    } else throw new Error(res.detail||'Xato');
  }catch(e){
    btn.textContent='✅ Saqlash'; btn.disabled=false;
    alert('Xato: '+e.message);
  }
}

// ── FILTER ────────────────────────────────────────────────
async function loadFilter(){
  const dateRaw = document.getElementById('filter-date').value;
  if(!dateRaw) return;
  const [y,m,d] = dateRaw.split('-');
  const date = `${d}.${m}.${y}`;
  const el = document.getElementById('filter-content');
  el.innerHTML='<div class="loader"><div class="spinner"></div>Yuklanmoqda...</div>';
  try{
    const data = await fetch(API+'/by-date?date='+encodeURIComponent(date)).then(r=>r.json());
    let html = `<div class="date-badge">📅 ${date}</div>`;
    if(data.chiqimlar.length){
      html += '<div class="section-title">Chiqimlar</div><div class="tx-list">';
      data.chiqimlar.forEach(t=>{ html += txItem(t,'CHIQIM',true); });
      html += '</div>';
    }
    if(data.kirimlar.length){
      html += '<div class="section-title">Kirimlar</div><div class="tx-list">';
      data.kirimlar.forEach(t=>{ html += txItem(t,'KIRIM',true); });
      html += '</div>';
    }
    if(!data.chiqimlar.length && !data.kirimlar.length){
      html += '<div class="empty"><div class="empty-icon">🔍</div>Bu sanada ma\'lumot yo\'q</div>';
    }
    el.innerHTML = html;
  }catch(e){ el.innerHTML='<div class="empty">❌ '+e.message+'</div>'; }
}

      },options:{plugins:{legend:{display:false}},cutout:'60%'}});

    if(ki.by_tur.length){

      new Chart(ctx2,{type:'doughnut',data:{

function openEdit(row, sheet, t){
  state.editRow = row; state.editSheet = sheet;
  const usd = t.usd>0?t.usd:'';
  const uzs = t.uzs>0?t.uzs:'';
  const valyuta = t.usd>0?'USD':'UZS';
  const summa = t.usd>0?t.usd:t.uzs;
  document.getElementById('edit-content').innerHTML = `
    <div class="form-group">
      <div class="label">Tur: <b>${EMOJIS[t.tur]||''} ${t.tur}</b> &nbsp; Egasi: <b>${t.egasi}</b></div>
    </div>
    <div class="form-group">
      <label class="form-label">Valyuta</label>
      <div class="btn-grid col2">
        <button class="sel-btn${valyuta==='USD'?' active':''}" onclick="selBtn('valyuta',this)">💵 USD</button>
        <button class="sel-btn${valyuta==='UZS'?' active':''}" onclick="selBtn('valyuta',this)">🇺🇿 UZS</button>
      </div>
    </div>
    <div class="form-group">
      <label class="form-label">Summa</label>
      <input type="number" id="e-summa" value="${summa}">
    </div>
    <div class="form-group">
      <label class="form-label">Note</label>
      <input type="text" id="e-note" value="${t.note||''}">
    </div>
    <button class="submit-btn" onclick="saveEdit()">💾 Saqlash</button>
  `;
  document.getElementById('edit-modal').classList.add('open');
}

async function saveEdit(){
  const summa = parseFloat(document.getElementById('e-summa').value);
  const note  = document.getElementById('e-note').value;
  const body  = {summa, note, valyuta: state.valyuta};
  try{
    const r = await fetch(`${API}/transaction/${state.editSheet}/${state.editRow}`,{
      method:'PUT', headers:{'Content-Type':'application/json'}, body:JSON.stringify(body)
    });
    const res = await r.json();
    if(res.success){ closeModal(); loadHistory(); loadToday(); }
    else throw new Error(res.detail);
  }catch(e){ alert('Xato: '+e.message); }
}

function closeModal(e){
  if(e && e.target!==document.getElementById('edit-modal')) return;
  document.getElementById('edit-modal').classList.remove('open');
}

// ── HELPERS ───────────────────────────────────────────────
function sumStr(usd, uzs){
  const parts=[];
  if(usd&&usd>0) parts.push(Math.round(usd)+'$');
  if(uzs&&uzs>0) parts.push(fmtN(uzs)+" so'm");
  return parts.join(' + ')||'0';
}
function fmtN(n){ return Math.round(n).toLocaleString('uz-UZ'); }

function txItem(t, sheet, editable=false){
  const sum = t.usd>0?Math.round(t.usd)+'$':fmtN(t.uzs)+" so'm";
  const color = sheet==='CHIQIM'?'red':'green';
  const badge = sheet==='CHIQIM'?'ch':'ki';
  const onclick = editable?`onclick="openEdit(${t.row},'${sheet}',${JSON.stringify(t).replace(/"/g,'&quot;')})"` : '';
  return `
    <div class="tx-item" ${onclick}>
      <div class="tx-left">
        <div class="tx-tur">${EMOJIS[t.tur]||''} ${t.tur}</div>
        <div class="tx-meta"><span class="badge ${badge}">${sheet}</span>${t.egasi} · ${t.sana||''}</div>
        ${t.note?`<div class="tx-note">${t.note}</div>`:''}
      </div>
      <div class="tx-right">
        <div class="tx-sum ${color}">${sheet==='CHIQIM'?'-':'+'}${sum}</div>
        <div class="tx-note">${t.tolov||''}</div>
      </div>
    </div>`;
}
